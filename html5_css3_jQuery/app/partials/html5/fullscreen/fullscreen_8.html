var open = document.querySelector("#open"); open.removeEventListener("click"); open.addEventListener("click", function(e){ if(document.webkitFullscreenEnabled) { document.onwebkitfullscreenchange = function(event) { if(document.querySelector(".text-content:-webkit-full-screen")) document.querySelector("button").innerHTML = "Fullscreen Mode"; else document.querySelector("button").innerHTML = "Normal Mode"; } document.onwebkitfullscreenerror = function(event) { //Error! } document.querySelector("div.text-content").webkitRequestFullscreen(); } });
