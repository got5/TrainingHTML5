A sample showing how to retrieve the children of a given element.

<div id="contenu"> <p>List</p> <ul> <li>List 1</li> <li>List 2</li> <li>List 3</li> </ul> </div> $('#displayChildren').off('click').on('click', function(){ jQuery("#contenu").children().each(function(index, element){ console.log(element.tagName); }); });
