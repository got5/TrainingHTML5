• Call the setItem method of the localStorage object
  • Store Strings only
  • Capacity : W3C recommend 5MB per origin
  • Sandboxed data
var button = document.getElementById("storage"); button.removeEventListener("click"); button.addEventListener("click", function(){ localStorage.setItem("test", "this is a test item"); //Or you can use the JS setter syntax //localStorage.test = "this is a test item"; });
