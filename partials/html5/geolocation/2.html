• Is the API available? Check if the geolocation field is available
  • Get the position?
    • At a specific time: geolocation.getCurrentPosition
    • Continuously: geolocation.watchPosition
    • Stop to check: geolocation.clearWatch
//Is the API available if(navigator.geolocation) {...} //Get position at a time, use getCurrentPosition() var myCallback= function(position){ console.log('latitude:' + position.coords.latitude); console.log('longitude:' + position.coords.longitude); console.log('accuracy:' + position.coords.accuracy); } geolocation.getCurrentPosition(myCallback); //Watch position continuously, use watchPosition() var myCallback= function(position){ console.log('latitude:' + position.coords.latitude); console.log('longitude:' + position.coords.longitude); console.log('accuracy:' + position.coords.accuracy); } //Start watching var watchId= geolocation.watchPosition(myCallback); //Stop watching geolocation.clearWatch(watchId);
