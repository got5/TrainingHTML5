• The idea is to give the ability to find DOM elements using CSS selectors
  • Previously, the standard way to access DOM elements was to use specific methods
    • One based on IDs
    • One based on tag names
    • ...
partial interface HTMLDocument {
	Element  getElementById(in DOMString elementId);
	NodeList getElementsByTagName(in DOMString name);
	NodeList getElementsByTagNameNS(in DOMString namespaceURI, in DOMString name);
	NodeList getElementsByClass(in DOMString class);
	NodeList getElementsByName(in DOMString name);
}

partial interface HTMLElement {
	NodeList getElementsByTagName(in DOMString name);
	NodeList getElementsByTagNameNS(in DOMString namespaceURI, in DOMString name) raises(DOMException);
	NodeList getElementsByClass(in DOMString class);
}

//Sample
var el = document.getElementById("elementId");
var inputs = el.getElementsByTagName("input");
