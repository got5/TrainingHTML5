• We can configure Require in the main.js with require.config() function
  • Librairies like JQuery run as a module (use module name and not path) when they are detecting Require.
require.config({ //By default load any module IDs from js/lib baseUrl: "js/lib", //except, if the module ID starts with "app", //load it from the js/app directory. "paths": { "app": "../app" } }); require(['jquery', 'app/myApp'], function($, myApp){ //do my job here });
