• We can configure Require in the main.js with require.config() function
  • Librairies like JQuery run as a module (use module name and not path) when they are detecting Require.
  • We can configure global variable.
require.config({ baseUrl: "js/lib", "paths": { "app": "../app", "jquery": "//ajax.googleapis.com/ajax/libs/jquery/2.0.0/jquery.min" }, "shim": { "jquery":{ "exports": "$" } } }); require(["app"]);
