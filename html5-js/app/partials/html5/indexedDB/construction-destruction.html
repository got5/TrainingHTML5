var openRequest = window.indexedDB.open('InitDB',1); //add suffix to variable to know which request is which //open(databasename,versionNumber) to create or access a DB openRequest.onupgradeneeded = function(e){ var newVersion = e.target.result; if(!newVersion.objectStoreNames.contains('course')){ newVersion.createObjectStore('courses', { autoIncrement:true }); } }; openRequest.onerror = openRequest.onblocked = console.log; openRequest.onsuccess = function(e){ console.log('Database opened'); initDB = e.target.result; }; function _deleteDatabase(){ console.log("check if the database exists"); if(typeof initDB !== 'undefined'){ console.log('Closing the database...'); initDB.close(); //db have to be closed to be deleted //This call is async ! console.log('try to delete the db'); var deleteRequest = indexedDB.deleteDatabase('InitDB'); deleteRequest.onerror = deleteRequest.onblocked = console.log; deleteRequest.onsuccess = function(){ console.log('Database Deleted'); } }} var indexedDBCtrl = function(){ this.deleteDatabase = _deleteDatabase; }; <!--This use the AngularJS framework.
