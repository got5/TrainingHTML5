• Use options to tune behavior
    • enableHighAccuracy: get a more precise position
    • timeout: Wait this time before calling the error function callback
    • maximumAge:keep position in cache this time instead of ask for the real position
  • Manage errors: check for the error type in the appropriated callback (timeout, permission denied, ...)
var whatToDo= function(position){ ... } var errorCallback= function(error) { switch(error.code) { case error.TIMEOUT: console.log('Timeout !'); break; case error.PERMISSION_DENIED: console.log('Permission denied !'); break; case error.POSITION_UNAVAILABLE: console.log('La position n’a pu être déterminée'); break; case error.UNKNOWN_ERROR: console.log('Erreur inconnue'); break; } } var opts = { enableHighAccuracy: true, maximumAge : 30000, timeout : 27000 }; geolocation.watchPosition(whatToDo, errorCallback, opts);
