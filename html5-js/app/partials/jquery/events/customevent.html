• Possibility to manually trigger an event
  • Can pass extra datas to Event Handlers
  • Creation of custom events

Trigger Custom Event

jQuery(document).ready(function(){ jQuery("#trigger").off("my_event"); jQuery("#trigger").off("click"); jQuery("#trigger").on("my_event", function(e, param1, param2){ e.preventDefault(); alert(param1 + " " + param2); }); jQuery("#trigger").on("click", function(e, param1, param2){ e.preventDefault(); jQuery("#trigger").trigger("my_event", ["Custom", "Event"]); }); });
