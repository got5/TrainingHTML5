Can be usefull with forms, for example to make sure the form is not sent depending on some validation rules

<form> <p> <label> name </label> <input type="text" /> <input type="submit" /> </p> </form> jQuery(document).ready(function(){ jQuery("form") .off("submit") .on("submit", function(event){ var isRuleOk= jQuery("input[type='text']").val().length > 0; if (!isRuleOk){ event.preventDefault(); alert('KO'); } //just an example, never submit... return false; }); });
