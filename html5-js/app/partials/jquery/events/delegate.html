• Efficient way to bind handlers on multiple elements
  • Can manage events triggered by not-yet-existing element
<div class="d1">1</div> <div class="d2">2</div> <div class="d3">3</div> jQuery(document).ready(function(){ jQuery(".bloc-content").off("click", ".d1, .d2, .d3"); jQuery(".bloc-content").on("click", ".d1, .d2, .d3", function(event){ alert(event.target.className); }); }); .d1 { background-color: green; } .d2 { background-color: blue; } .d3 { background-color: red; }
