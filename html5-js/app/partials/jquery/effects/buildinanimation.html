• You can easily use show(), hide() or toggle() on an element
jQuery(document).ready(function(){ jQuery("#hide").off("click").on("click", function(event){ jQuery("pre#editor").hide(); }); jQuery("#show").off("click").on("click", function(event){ jQuery("pre#editor").show(); }); jQuery("#toggle").off("click").on("click", function(event){ jQuery("pre#editor").toggle(); }); });
