• Specify the speed at which the animation progresses at different points within the animation
  • The default value is swing: slower at start and at end
  • Only Two implementations in jQuery : swing and linear
  • More easing functions are available with the use of plugins
  • We can specify an easing function per property
jQuery(document).ready(function(){ jQuery('#anime1') .off('click').on('click', function(event){ event.preventDefault(); jQuery('pre#editor') .animate({'height': '-=100'}, 'slow', 'linear') .animate({'height': '+=100'}, 'slow', 'linear'); }); }); jQuery(document).ready(function(){ //opacity uses 'linear' while height property uses 'swing' jQuery('#anime2') .off('click').on('click', function(event){ event.preventDefault(); jQuery('pre#editor') .animate( { 'opacity': 0, 'height': ['500px', 'swing'], },'slow', 'linear') .animate( { 'opacity': 1, 'height': ['210px', 'swing'], },'slow', 'linear'); }); });
