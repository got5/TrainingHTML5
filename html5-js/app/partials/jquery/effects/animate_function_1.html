• Possibility to animate a value relative to its original value
  • Use += or -= to increase or decrease the value
  • You can animate by pixels, ems, percentage, ...
jQuery(document).ready(function(){ jQuery("#anime").off("click"); jQuery("#anime").on("click", function(event){ event.preventDefault(); jQuery("pre#editor").animate({ "height": "+=200px" }); /* jQuery("pre#editor").animate({ "height": "+=10%" }); */ }); });
