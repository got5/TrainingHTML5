• The animate() method is used to animate a number of CSS properties on an element over a period of time
  • Most properties that are non-numeric cannot be animated
  • Speed in millisecond (or slow, normal, fast)
  • To override default speed : jQuery.fx.speeds.slow = 500;
  • Possibility to execute a callback when the animation is finished
  • All jQuery effects can be turned off globally by setting jQuery.fx.off = true
jQuery(document).ready(function(){ jQuery("#anime").off("click"); jQuery("#anime").on("click", function(event){ event.preventDefault(); jQuery("pre#editor").animate({ "height": 5 }, 500, function(){alert("the animation is finished!")}); }); });
